t-test HYPOTHESIS

In[23]:
from scipy import stats

# Degrees of freedom
df = X_train.shape[0] - X_train.shape[1]

# Residual standard error
mse = np.mean(mlr_diff['Residuals ']**2)
rse = np.sqrt(mse)

# Standard error of each coefficient
X_with_intercept = np.hstack((np.ones((X_train.shape[0], 1)), X_train))  # Add a column of ones for the intercept
X_inv = np.linalg.inv(np.dot(X_with_intercept.T, X_with_intercept))
se = rse * np.sqrt(np.diag(X_inv))

# t-statistics for each coefficient
t_stats = mlr_model.coef_ / se[1:]  # Exclude the intercept standard error

# P-values for each coefficient
p_values = 2 * (1 - stats.t.cdf(np.abs(t_stats), df))
p_values
